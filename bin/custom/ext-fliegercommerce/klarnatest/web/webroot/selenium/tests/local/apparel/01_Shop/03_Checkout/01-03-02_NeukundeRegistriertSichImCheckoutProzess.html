01-03-02 Neukunde Registriert Sich im Checkout Prozess
ge_tearUpTC 01-03-02 Neukunde Registriert Sich im Checkout Prozess
ge_setTCDescription Kunde aus ${Land} mit dem Klarna Status ${Klarna_Status} und der Klarna Zahlungsart ${Klarna_Selected_Payment_Method} befindet sich auf der Registrierungsseite und möchte sich neu registrieren Kunde aus  ${Land} mit den Login-Daten: Benutzername ${Neukundenregistrierung_Email} und Passwort ${Neukundenregistrierung_Kennwort} hat sich neu registriert
assertElementPresent id=register.title
assertElementPresent id=register.firstName
assertElementPresent id=register.lastName
assertElementPresent id=register.email
assertElementPresent id=password
assertElementPresent id=register.checkPwd
select id=register.title label=${Lieferadresse_Anrede}
type id=register.firstName ${Neukundenregistrierung_Vorname}
type id=register.lastName ${Neukundenregistrierung_Nachname}
type id=register.email ${Neukundenregistrierung_Email}
ge_setTCParameter ${Neukundenregistrierung_Email} Neukunde_Username
type id=password ${Neukundenregistrierung_Kennwort}
ge_setTCParameter ${Neukundenregistrierung_Kennwort} Neukunde_Passwort
type id=register.checkPwd ${Neukundenregistrierung_Kennwort}
assertElementPresent css=button.form
clickAndWait css=button.form
pause 3000
storeElementPresent css=span.form_field_error isPresent
echo ${isPresent}
gotoIf ${isPresent} == false REGISTRIERUNG_ERFOLGT
type id=j_username ${Neukundenregistrierung_Email}
type id=j_password ${Neukundenregistrierung_Kennwort}
clickAndWait css=#loginForm > span > button.form
label REGISTRIERUNG_ERFOLGT
ge_tearDownTC
