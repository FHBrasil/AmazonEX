03-01-01 Oeffne Klarna Testsystem Per URL
ge_tearUpTC 03-01-01 Oeffne Klarna Testsystem
ge_setTCDescription Öffnen des Klarna Betatestsystems Das Klarna Betatestsystem ist erreichbar und dessen Startseite ist geöffnet worden
open ${KLARNA_TESTSYSTEM_URL}
pause 2000
assertTitle ${KlarnaTestsystemSeitentitel} ${KlarnaTestsystemSeitentitel}
assertElementPresent css=img[alt="German"]
clickAndWait css=img[alt="German"]
assertElementPresent id=uidid
assertElementPresent name=password
type id=uidid ${KLARNA_TESTSYSTEM_USER}
type name=password ${KLARNA_TESTSYSTEM_PW}
assertElementPresent name=submit
clickAndWait name=submit
pause 2000
assertText css=h1 Willkommen auf Klarna Online
ge_tearDownTC
