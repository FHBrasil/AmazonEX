01-03-08 Neukunde Registriert Sich Ausserhalb des Checkout Prozesses
ge_tearUpTC 01-03-08 Neukunde Registriert Sich Ausserhalb des Checkout Prozesses
ge_designateTCParameters {TCVariableName1},{TCVariableName2}
ge_setTCDescription Kunde aus ${Land} befindet sich auf der Anmeldungs-/Registrierungsseite und möchte sich neu registrieren Kunde aus  ${Land} mit den Login-Daten: Benutzername ${Datumsstempel}_${Neukundenregistrierung_Email} und Passwort ${Datumsstempel}_${Neukundenregistrierung_Kennwort} hat sich neu registriert
assertElementPresent id=register.title
assertElementPresent id=register.firstName
assertElementPresent id=register.lastName
assertElementPresent id=register.email
assertElementPresent id=password
assertElementPresent id=register.checkPwd
select id=register.title label=Herr
type id=register.firstName ${Datumsstempel}_${Neukundenregistrierung_Vorname}
type id=register.lastName ${Datumsstempel}_${Neukundenregistrierung_Nachname}
type id=register.email ${Datumsstempel}_${Neukundenregistrierung_Email}
ge_setTCParameter ${Datumsstempel}_${Neukundenregistrierung_Email} Neukunde_Username
type id=password ${Datumsstempel}_${Neukundenregistrierung_Kennwort}
ge_setTCParameter ${Datumsstempel}_${Neukundenregistrierung_Kennwort} Neukunde_Passwort
type id=register.checkPwd ${Datumsstempel}_${Neukundenregistrierung_Kennwort}
assertElementPresent css=button.form
clickAndWait css=button.form
ge_tearDownTC
